CREATE TABLE Departments (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    DepartmentName NVARCHAR(MAX) NULL,
	Code NVARCHAR(20)
);
GO

CREATE TABLE Students (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Name NVARCHAR(MAX) NULL,
    AdmissionDate DATETIME2 NOT NULL DEFAULT GETDATE(),
    IsActive BIT NOT NULL,
    RegistrationFee DECIMAL(18,2) NOT NULL,
    ImageUrl NVARCHAR(MAX) NULL,
    DepartmentId INT NOT NULL REFERENCES Departments(Id)
);
GO

CREATE TABLE Subjects (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Course NVARCHAR(MAX) NULL,
    Credit DECIMAL(18,2) NOT NULL,
    StudentId INT NOT NULL REFERENCES Students(Id) ON DELETE CASCADE
);
GO

using System;
using System.Collections.Generic;
using System.ComponentModel.DataAnnotations;   // ← this is for DataAnnotaions
using System.Web;
using System.Web.Mvc;
namespace StudentMasterDetails.Models
{
    public class StudentVM
    {
        public int Id { get; set; }

        [Required, StringLength(100), Display(Name = "Full Name")]
        public string Name { get; set; }

        [Required, DataType(DataType.Date), Display(Name = "Admission Date")]
        public DateTime AdmissionDate { get; set; } = DateTime.Now;

        [Display(Name = "Active")]
        public bool IsActive { get; set; }

        [Display(Name = "Fee"), Range(0, 999999)]
        public decimal RegistrationFee { get; set; }

        public string ImageUrl { get; set; }

        [Required, Display(Name = "Department")]
        public int DepartmentId { get; set; }

        public HttpPostedFileBase Picture       { get; set; }
        public SelectList         DepartmentList{ get; set; }
        public List<Subject>      Subjects      { get; set; } = new List<Subject>();
    }
}

//(4) Controllers=>MVC 5 Empty=>StudentsController

using StudentMasterDetails.Models;
using System; using System.Data.Entity; using System.IO;
using System.Linq; using System.Web.Mvc;

namespace StudentMasterDetails.Controllers
{
    public class StudentsController : Controller
    {
        private StudentMasterDetailsDBEntities db =
            new StudentMasterDetailsDBEntities();

        // ★ Dropdown helper
        SelectList GetDepts(int? sel = null) =>
            new SelectList(db.Departments.OrderBy(d => d.DepartmentName).ToList(),
                           "Id", "DepartmentName", sel);

        // INDEX
        public ActionResult Index() =>
            View(db.Students.Include(s => s.Department).ToList());

        // CREATE GET
        public ActionResult Create() =>
            View(new StudentVM { DepartmentList = GetDepts() });

        // CREATE POST
        [HttpPost, ValidateAntiForgeryToken]
        public ActionResult Create(StudentVM vm)
        {
            if (ModelState.IsValid)
            {
                var s = new Student {
                    Name = vm.Name, AdmissionDate = vm.AdmissionDate,
                    IsActive = vm.IsActive, RegistrationFee = vm.RegistrationFee,
                    DepartmentId = vm.DepartmentId };

                // ★ Image — 3-line pattern
                if (vm.Picture != null && vm.Picture.ContentLength > 0) {
                    string fn = Path.GetFileNameWithoutExtension(vm.Picture.FileName)
                                + "_" + DateTime.Now.Ticks
                                + Path.GetExtension(vm.Picture.FileName);
                    vm.Picture.SaveAs(Server.MapPath("~/Images/" + fn));
                    s.ImageUrl = "/Images/" + fn;
                }
                // ★ Save subjects
                if (vm.Subjects != null)
                    foreach (var sub in vm.Subjects)
                        if (!string.IsNullOrWhiteSpace(sub.Course))
                            s.Subjects.Add(new Subject { Course=sub.Course, Credit=sub.Credit });

                db.Students.Add(s);
                db.SaveChanges();
                return RedirectToAction("Index");
            }
            vm.DepartmentList = GetDepts(vm.DepartmentId);
            return View(vm);
        }

        // EDIT GET
        public ActionResult Edit(int id)
        {
            var s = db.Students.Find(id);
            if (s == null) return HttpNotFound();
            return View(new StudentVM {
                Id=s.Id, Name=s.Name, AdmissionDate=s.AdmissionDate,
                IsActive=s.IsActive, RegistrationFee=s.RegistrationFee,
                ImageUrl=s.ImageUrl, DepartmentId=s.DepartmentId,
                DepartmentList=GetDepts(s.DepartmentId),
                Subjects=s.Subjects.ToList() });
        }

        // EDIT POST
        [HttpPost, ValidateAntiForgeryToken]
        public ActionResult Edit(StudentVM vm)
        {
            if (ModelState.IsValid)
            {
                var s = db.Students.Find(vm.Id);
                s.Name=vm.Name; s.AdmissionDate=vm.AdmissionDate;
                s.IsActive=vm.IsActive; s.RegistrationFee=vm.RegistrationFee;
                s.DepartmentId=vm.DepartmentId;

                // ★ Image — same 3-line pattern
                if (vm.Picture != null && vm.Picture.ContentLength > 0) {
                    string fn = Path.GetFileNameWithoutExtension(vm.Picture.FileName)
                                + "_" + DateTime.Now.Ticks
                                + Path.GetExtension(vm.Picture.FileName);
                    vm.Picture.SaveAs(Server.MapPath("~/Images/" + fn));
                    s.ImageUrl = "/Images/" + fn;
                }
                // ★ Replace all subjects
                db.Subjects.Where(x => x.StudentId == s.Id).ToList()
                           .ForEach(x => db.Subjects.Remove(x));
                if (vm.Subjects != null)
                    foreach (var sub in vm.Subjects)
                        if (!string.IsNullOrWhiteSpace(sub.Course))
                            s.Subjects.Add(new Subject { Course=sub.Course, Credit=sub.Credit });

                db.SaveChanges();
                return RedirectToAction("Index");
            }
            vm.DepartmentList = GetDepts(vm.DepartmentId);
            return View(vm);
        }

        // DETAILS
        public ActionResult Details(int id)
        {
            var s = db.Students.Include(x => x.Department).SingleOrDefault(x => x.Id == id);
            return s == null ? HttpNotFound() : (ActionResult)View(s);
        }

        // DELETE POST
        [HttpPost, ValidateAntiForgeryToken]
        public ActionResult Delete(int id)
        {
            var s = db.Students.Find(id);
            if (s != null) { db.Students.Remove(s); db.SaveChanges(); }
            return RedirectToAction("Index");
        }

        // ★★ MVC AJAX endpoint
        public JsonResult GetStudentCount() =>
            Json(new { count = db.Students.Count() }, JsonRequestBehavior.AllowGet);

        protected override void Dispose(bool d)
        { if (d) db.Dispose(); base.Dispose(d); }
    }
}
