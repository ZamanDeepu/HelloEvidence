using System;
using System.Collections.Generic;
using System.ComponentModel.DataAnnotations;
using System.ComponentModel.DataAnnotations.Schema;
using System.Data.Entity;
using System.Linq;
using System.Web;

namespace StudentManagementSystemMvcWebApi.Models
{
    public class Department  // Master table for dropdown
    {
        public int Id { get; set; }
        [Required, StringLength(50)]
        public string Name { get; set; }
        public virtual ICollection<Student> Students { get; set; } = new List<Student>();
    }

    public class Student
    {
        public int Id { get; set; }
        [Required, StringLength(50)]
        public string Name { get; set; }              // Text
        public DateTime Date { get; set; } = DateTime.Now;   // Date
        public bool Activity { get; set; } = true;    // Boolean
        public string Image { get; set; }              // Image (filename)
        [ForeignKey("Department")]
        public int DepartmentId { get; set; }           // Relation data (dropdown)
        public virtual Department Department { get; set; }
        public virtual ICollection<Subject> Subjects { get; set; } = new List<Subject>(); // Details table
    }

    public class Subject
    {
        public int Id { get; set; }
        [Required(ErrorMessage = "Course Required"), StringLength(100)]
        public string Course { get; set; }
        public int Credit { get; set; }
        [ForeignKey("Student")]
        public int StudentId { get; set; }              // Relation to Student
        public virtual Student Student { get; set; }
    }
}



//Step 02: Models/IdentityModels.cs


public class ApplicationDbContext : IdentityDbContext<ApplicationUser>
{

public ApplicationDbContext() : base("DefaultConnection", throwIfV1Schema: false) { }


public DbSet<Department> Departments { get; set; }
public DbSet<Student> Students { get; set; }
public DbSet<Subject> Subjects { get; set; }

public static ApplicationDbContext Create() { return new ApplicationDbContext(); }

}

//Step 06: Controllers/StudentsController.cs


using Newtonsoft.Json;
using StudentManagementSystemMvcWebApi.DTOs;
using StudentManagementSystemMvcWebApi.Models;
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Net;
using System.Net.Http;
using System.Web;
using System.Web.Http;
using System.Data.Entity;

namespace StudentManagementSystemMvcWebApi.Controllers
{
    [Authorize]
    public class StudentsController : ApiController
    {
        ApplicationDbContext db = new ApplicationDbContext();

        [HttpGet]
        [Route("api/students")]
        public IHttpActionResult GetStudents()
        {
            var students = db.Students
                .Include(d=>d.Department)
                .Include(d=>d.Subjects)
                .Select(d=>new StudentDto
                {
                    Id = d.Id,
                    Name = d.Name,
                    Date = d.Date,
                    Activity=d.Activity,
                    Image=d.Image,
                    DepartmentId=d.DepartmentId,
                    DepartmentName=d.Department.Name,
                    Subjects=d.Subjects.Select(sub=> new SubjectDto
                    {
                        Id = sub.Id,
                        Course = sub.Course,
                        Credit = sub.Credit,
                    }).ToList()
                }).ToList();
            return Ok(students);
        }

        [HttpGet]
        [Route("api/students/{id}")]
        public IHttpActionResult GetStudent(int id)
        {
            var s = db.Students
                .Include(x => x.Department)
                .Include(x => x.Subjects)
                .FirstOrDefault(x => x.Id == id);

            if (s == null) return NotFound();

            var dto = new StudentDto
            {
                Id = s.Id,
                Name = s.Name,
                Date = s.Date,
                Activity = s.Activity,
                Image = s.Image,
                DepartmentId = s.DepartmentId,
                DepartmentName = s.Department?.Name,
                Subjects = s.Subjects.Select(sub => new SubjectDto
                {
                    Id = sub.Id,
                    Course = sub.Course,
                    Credit = sub.Credit,
                }).ToList()
            };
            return Ok(dto);
        }

        [HttpPost]
        [Route("api/students")]
        public IHttpActionResult PostStudent([FromBody] StudentDto model)
        {
            if (model == null) return BadRequest("Model is null");
            if (!ModelState.IsValid) return BadRequest(ModelState);

            var student = new Student
            {
                Name = model.Name,
                Date = model.Date == default(DateTime) ? DateTime.Now : model.Date,
                Activity = model.Activity,
                Image = model.Image,
                DepartmentId = model.DepartmentId
            };

            // JSON serialize/deserialize convert
            if (model.Subjects != null && model.Subjects.Any())
            {
                var subjects = JsonConvert.DeserializeObject<List<Subject>>(
                                JsonConvert.SerializeObject(model.Subjects));

                foreach (var s in subjects)
                {
                    s.Id = 0;          // reset PK
                    s.StudentId = 0;   // let EF set FK
                    student.Subjects.Add(s);
                }
            }
            db.Students.Add(student);
            db.SaveChanges();
            return Ok(student);
        }

        [HttpPut]
        [Route("api/students/{id}")]
        public IHttpActionResult PutStudent(int id, [FromBody] StudentDto model)
        {
            if (model == null) return BadRequest("Model is null");
            if (!ModelState.IsValid) return BadRequest(ModelState);

            var student = db.Students
                            .Include(s => s.Subjects)
                            .FirstOrDefault(s => s.Id == id);

            if (student == null) return NotFound();

            student.Name = model.Name;
            student.Date = model.Date == default(DateTime) ? DateTime.Now : model.Date;
            student.Activity = model.Activity;
            student.Image = model.Image;
            student.DepartmentId = model.DepartmentId;

            db.Subjects.RemoveRange(student.Subjects);      // Remove old subjects
            student.Subjects.Clear();

            if (model.Subjects != null && model.Subjects.Any())
            {
                var subjects = JsonConvert.DeserializeObject<List<Subject>>(
                                JsonConvert.SerializeObject(model.Subjects));

                foreach (var s in subjects)
                {
                    s.Id = 0;             // reset PK
                    s.StudentId = 0;      // let EF set FK
                    student.Subjects.Add(s);
                }
            }
            db.SaveChanges();
            return Ok(student);
        }

        [HttpDelete, Route("api/students/{id}")]
        public IHttpActionResult DeleteStudent(int id)
        {
            var student = db.Students
                .Include(s => s.Subjects)
                .FirstOrDefault(s => s.Id == id);

            if (student == null) return NotFound();

            db.Subjects.RemoveRange(student.Subjects);
            db.Students.Remove(student);
            db.SaveChanges();
            return Ok(new { message = $"Student {id} and their subjects were deleted." });
        }

        [HttpPost]
        [Route("api/students/{id}/images")]
        public IHttpActionResult UploadImage(int id)
        {
            var student = db.Students.FirstOrDefault(s => s.Id == id);
            if (student == null) return NotFound();

            var file = HttpContext.Current.Request.Files[0];
            if (file == null || file.ContentLength == 0)
                return BadRequest("No file uploaded.");

            var folderPath = HttpContext.Current.Server.MapPath("~/Images");
            if (!Directory.Exists(folderPath))
                Directory.CreateDirectory(folderPath);

            var fileName = $"{id}_{Guid.NewGuid()}{Path.GetExtension(file.FileName)}";
            file.SaveAs(Path.Combine(folderPath, fileName));

            student.Image = fileName;
            db.SaveChanges();
            return Ok(new { image = student.Image });
        }
    }
}
